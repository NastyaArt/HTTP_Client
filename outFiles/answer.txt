              var curCookie = name + "=" + escape(value) +
                ((expires) ? "; expires=" + expires.toGMTString() : "") +
                ((path) ? "; path=" + path : "; path=/") +
                ((domain) ? "; domain=" + domain : "") +
                ((secure) ? "; secure" : "");
              document.cookie = curCookie;
            }
            function getCookie(name) {
              var prefix = name + "=";
              var cookieStartIndex = document.cookie.indexOf(prefix);
              if(cookieStartIndex == -1) return null;
              var cookieEndIndex = document.cookie.indexOf(";", cookieStartIndex + prefix.length);
              if(cookieEndIndex == -1) cookieEndIndex = document.cookie.length;
              return unescape(document.cookie.substring(cookieStartIndex + prefix.length, cookieEndIndex));

<h2>Тестирование Денвера</h2>

<p>Настоятельно рекомендуем проверить работоспособность сервера при помощи следующих далее ссылок. Посмотрев результат, нажмите кнопку <b>Назад</b>, чтобы перейти к другой ссылке. Если что-то не заработало, проделайте следующие действия:</p>
<ol>
<li>Загляните в log-файл сервера (<tt>/usr/local/apache/logs/error.log</tt>).</li>
<li>Прочитайте <a href=http://www.denwer.ru/faq/>ответы на часто задаваемые вопросы</a>.</li>
<li>Если ответа нет в FAQ, зайдите в форум по адресу <a href=http://forum.dklab.ru/denwer/>http://forum.dklab.ru/denwer</a> и воспользуйтесь <a href=http://forum.dklab.ru/search.php>поисковой системой</a> — 9 шансов из 10, что ответ на ваш вопрос уже есть в базе данных.</li>
</ol>

<p><font color="red"><b>Внимание</b></font> — распространенная ошибка: если работает только <tt>localhost</tt>, а другие хосты — <i>не работают</i>, пожалуйста, проверьте, выключен ли у Вас прокси-сервер (<b>Сервис — Свойства обозревателя — Подключение — Настройка сети</b>).</p>

<p>Пожалуйста, описывайте проблему как можно подробнее. Помните: каждое ваше письмо и наш ответ на него попадают в глобальный архив изменений (<tt>CHANGELOG</tt>), а потом и в FAQ.  Чем более исчерпывающим будет описание проблемы, тем оперативнее придет ответ, и тем больше пользы принесет ваше письмо другим Web-разработчикам.</p>
